open import Data.Product
open import Relation.Binary.Construct.Closure.ReflexiveTransitive
open import Relation.Binary

open import Stoughton.Var

module ParallelReduction (𝒞 : Set) {𝒱 : Set} (var : Enum 𝒱) where

  open import Stoughton.Syntax 𝒞 𝒱 (Enum._≟_ var)  
  open import Stoughton.Substitution 𝒞 var      
  open import Beta 𝒞 var  
  open import BetaReduction 𝒞 var
  
  private
    _≟_ = Enum._≟_ var
  
  infixl 3 _⇉_
  data _⇉_ : Λ → Λ → Set where
    ⇉c : {k : 𝒞} → c k ⇉ c k
    ⇉v : {x : 𝒱} → v x ⇉ v x
    ⇉λ : ∀ {x A A' M M'} → A ⇉ A' → M ⇉ M' → λ[ x ∶ A ] M ⇉ λ[ x ∶ A' ] M'
    ⇉Π : {x : 𝒱}{A A' M M' : Λ} → A ⇉ A' → M ⇉ M' → Π[ x ∶ A ] M ⇉ Π[ x ∶ A' ] M'    
    ⇉· : {M M' N N' : Λ} → M ⇉ M' → N ⇉ N' → M · N ⇉ M' · N'
    ⇉β : {x : 𝒱}{A M M' N N' : Λ} → M ⇉ M' → N ⇉ N' → (λ[ x ∶ A ] M) · N ⇉ M' [ x := N' ]

  ⇉ρ : Reflexive _⇉_
  ⇉ρ {c k} = ⇉c 
  ⇉ρ {v x} = ⇉v
  ⇉ρ {M · N} = ⇉· ⇉ρ ⇉ρ
  ⇉ρ {λ[ x ∶ A ] M} = ⇉λ ⇉ρ ⇉ρ
  ⇉ρ {Π[ x ∶ A ] M} = ⇉Π ⇉ρ ⇉ρ

  manyStepBetaContainsPar :  _⇉_ ⇒ _→β*₀_
  manyStepBetaContainsPar ⇉c           = ε  
  manyStepBetaContainsPar ⇉v           = ε
  manyStepBetaContainsPar (⇉λ A⇉B M⇉N) = abs-star-ty (manyStepBetaContainsPar A⇉B) ◅◅ abs-star (manyStepBetaContainsPar M⇉N)
  manyStepBetaContainsPar (⇉Π A⇉B M⇉N) =
    pi-star-ty (manyStepBetaContainsPar A⇉B) ◅◅ pi-star (manyStepBetaContainsPar M⇉N)
  manyStepBetaContainsPar (⇉· {M} {M'} {N} {N'} M⇉M' N⇉N') =
    app-star-l (manyStepBetaContainsPar M⇉M') ◅◅ app-star-r (manyStepBetaContainsPar N⇉N')
  manyStepBetaContainsPar (⇉β {x} {A} {M} {M'} {N} {N'} M⇉M' N⇉N') =
    app-star-l (abs-star (manyStepBetaContainsPar M⇉M')) ◅◅ app-star-r (manyStepBetaContainsPar N⇉N') ◅◅ →cxt β ◅ ε

  manyStepBetaContainsStarPar : Star _⇉_ ⇒ _→β*₀_
  manyStepBetaContainsStarPar ε = ε
  manyStepBetaContainsStarPar (M⇉N ◅ N⇉*P) = manyStepBetaContainsPar M⇉N ◅◅ manyStepBetaContainsStarPar N⇉*P

  parContainsOneStep : _→β_ ⇒ _⇉_
  parContainsOneStep (→cxt β) = ⇉β ⇉ρ ⇉ρ
  parContainsOneStep (→λR M→N) = ⇉λ ⇉ρ (parContainsOneStep M→N)
  parContainsOneStep (→ΠR M→N) = ⇉Π ⇉ρ (parContainsOneStep M→N)
  parContainsOneStep (→λL M→N) = ⇉λ (parContainsOneStep M→N) ⇉ρ
  parContainsOneStep (→ΠL M→N) = ⇉Π (parContainsOneStep M→N) ⇉ρ  
  parContainsOneStep (→·R M→N) = ⇉· ⇉ρ (parContainsOneStep M→N)
  parContainsOneStep (→·L M→N) = ⇉· (parContainsOneStep M→N) ⇉ρ 

  starParContainsManyStep : _→β*₀_ ⇒ Star _⇉_
  starParContainsManyStep ε = ε
  starParContainsManyStep (M→N ◅ N→*P) = parContainsOneStep M→N ◅ starParContainsManyStep N→*P

  manyStepBetaEqualsStarPar : _→β*₀_ ⇔ Star _⇉_
  manyStepBetaEqualsStarPar = starParContainsManyStep , manyStepBetaContainsStarPar
